Gamma

Try gamma function

for ( i <- 1 until 10 ) {
  val i_1:Int = i - 1
  println(s"\tΓ($i):$i_1! => ${slash.gamma(i.toDouble)} : ${slash.Factorial(i_1)}")
}
// 	Γ(1):0! => 1.0 : 1
// 	Γ(2):1! => 1.0 : 1
// 	Γ(3):2! => 2.0 : 2
// 	Γ(4):3! => 6.0 : 6
