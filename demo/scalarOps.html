Vector Scalar Operations

Vector Scalar Operations

Example

import ai.dragonfly.math.vector.*

val v1 = Vec[2](1.5, 2.5)
// v1: Vec[2] = Array(1.5, 2.5)
v1 + 2.0
// res0: Vec[2] = Array(3.5, 4.5)
v1 - 2.0
// res1: Vec[2] = Array(-0.5, 0.5)
v1 * 2.0
// res2: Vec[2] = Array(3.0, 5.0)
v1 / 2.0
// res3: Vec[2] = Array(0.75, 1.25)

v1.unary_-
// res4: Vec[2] = Array(-1.5, -2.5)

// mutating (no copy) methods
Vec[2](1.5, 2.5) += 2.5
// res5: Vec[2] = Array(4.0, 5.0)
Vec[2](1.5, 2.5) -= 2.5
// res6: Vec[2] = Array(-1.0, 0.0)
Vec[2](1.5, 2.5) *= 2.5
// res7: Vec[2] = Array(3.75, 6.25)
Vec[2](1.5, 2.5) /= 2.5
// res8: Vec[2] = Array(0.6000000000000001, 1.0)

//
v1.clampedMin(2.0)
// res9: Vec[2] = Array(2.0, 2.5)
v1.clampedMAX(2.0)
// res10: Vec[2] = Array(1.5, 2.0)

val v2 = Vec[3](1.5, 2.5, 3.5)
// v2: Vec[3] = Array(1.5, 2.5, 3.5)
v2.clamped(2.0, 3.0)
// res11: Vec[3] = Array(2.0, 2.5, 3.0)

// mutating (no copy) versions
Vec[2](1.5, 2.5).clampMin(2.0)
// res12: Vec[2] = Array(2.0, 2.5)
Vec[2](1.5, 2.5).clampMAX(2.0)
// res13: Vec[2] = Array(1.5, 2.0)
v2.copy.clamp(2.0, 3.0)
// res14: Vec[3] = Array(2.0, 2.5, 3.0)
